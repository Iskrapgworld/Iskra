const board = document.getElementById('board'); const hpEl = document.getElementById('hp'); const shieldEl = document.getElementById('shield'); const enemyHpEl = document.getElementById('enemyHp'); const enemyShieldEl = document.getElementById('enemyShield'); const timerBar = document.getElementById('timerBar'); const overlay = document.getElementById('overlay'); const turnText = document.getElementById('turnText'); const logBox = document.getElementById('battleLog'); const colors = ['red','green','blue','yellow','purple','orange']; let hp = 200, shield = 100, enemyHp = 500, enemyShield = 300; let grid = [], dragging = null, turnTimer, isPlayerTurn = true; let playerName = "Игрок", totalComboCount = 0; function startGame() { playerName = document.getElementById('nickname').value || "Игрок"; overlay.style.display = 'none'; document.body.classList.remove('enemy-turn'); hp = 200; shield = 100; enemyHp = 500; enemyShield = 300; updateStats(); createGrid(); startTurn(); } function updateStats() { hpEl.textContent = hp; shieldEl.textContent = shield; enemyHpEl.textContent = enemyHp; enemyShieldEl.textContent = enemyShield; } function createGrid() { board.innerHTML = ''; grid = []; for(let i=0;i<36;i++) { const cell = document.createElement('div'); cell.className = 'cell ' + colors[Math.floor(Math.random()*colors.length)]; cell.style.transition = 'transform 0.2s, opacity 1s ease, top 0.4s'; board.appendChild(cell); grid.push(cell); } } function index(r,c){ return r*6 + c; } function getColor(el){ return colors.find(c => el.classList.contains(c)); } function startTurn() { isPlayerTurn = true; totalComboCount = 0; document.body.classList.remove('enemy-turn'); board.style.pointerEvents = 'auto'; timerBar.style.width = '100%'; let time = 20; clearInterval(turnTimer); turnTimer = setInterval(() => { time--; timerBar.style.width = (time*5) + '%'; if(time <= 0){ clearInterval(turnTimer); board.style.pointerEvents = 'none'; cascadeAndApply(); } }, 1000); } function cascadeAndApply() { let matches = checkMatches(); const counts = {red:0,green:0,blue:0,orange:0,purple:0,yellow:0}; if(matches.length === 0){ if (isPlayerTurn) { setTimeout(() => monsterTurn(), 800); } else { setTimeout(() => startTurn(), 800); } return; } totalComboCount += matches.length; matches.forEach(g => { let color = getColor(grid[g[0]]); counts[color] += g.length; g.forEach(i => grid[i].classList.add('highlight')); }); setTimeout(() => { matches.flat().forEach(i => { grid[i].classList.remove('highlight'); grid[i].classList.add('fade'); }); setTimeout(() => { matches.flat().forEach(i => grid[i].className = 'cell'); dropDown(); applyEffects(counts, totalComboCount); setTimeout(() => cascadeAndApply(), 800); }, 1000); }, 1500); } function applyEffects(counts, comboCount) { let baseDamage = (counts.red + counts.purple) * 10; let bonusPercent = comboCount * 5; let dmg = Math.floor(baseDamage * (1 + bonusPercent / 100)); let heal = counts.green * 5 + (counts.green >= 5 ? 20 : 0); let def = counts.blue * 5; let critChance = Math.min(counts.orange * 5, 100); let isCrit = Math.random() * 100 < critChance; if (isCrit) dmg *= 2; if (isPlayerTurn) { let toShield = Math.floor(dmg / 2); let toHp = dmg - toShield; if (enemyShield >= toShield) { enemyShield -= toShield; } else { let remain = toShield - enemyShield; enemyShield = 0; enemyHp -= remain; } enemyHp -= toHp; if (enemyHp < 0) enemyHp = 0; shield += def; hp += heal; if (hp > 200) hp = 200; showLog(playerName, comboCount, dmg, isCrit, critChance, def, heal, true); } else { let toShield = Math.floor(dmg / 2); let toHp = dmg - toShield; if (shield >= toShield) { shield -= toShield; } else { let remain = toShield - shield; shield = 0; hp -= remain; } hp -= toHp; if (hp < 0) hp = 0; enemyShield += def; if (enemyShield > 300) enemyShield = 300; enemyHp += heal; if (enemyHp > 500) enemyHp = 500; showLog("Монстр", comboCount, dmg, isCrit, critChance, def, heal, false); } updateStats(); } function dropDown() { for (let c = 0; c < 6; c++) { for (let r = 5; r > 0; r--) { let i = index(r, c); if (!getColor(grid[i])) { for (let k = r - 1; k >= 0; k--) { let up = index(k, c); if (getColor(grid[up])) { grid[i].className = grid[up].className; grid[up].className = 'cell'; break; } } } } for (let r = 0; r < 6; r++) { let i = index(r, c); if (!getColor(grid[i])) { let cls = colors[Math.floor(Math.random() * colors.length)]; grid[i].className = 'cell ' + cls; } } } } function checkMatches() { const matches = []; for (let r = 0; r < 6; r++) { for (let c = 0; c < 4; c++) { let i = index(r, c); let a = grid[i], b = grid[i + 1], c_ = grid[i + 2]; if (getColor(a) === getColor(b) && getColor(b) === getColor(c_)) matches.push([i, i + 1, i + 2]); } } for (let c = 0; c < 6; c++) { for (let r = 0; r < 4; r++) { let i = index(r, c); let a = grid[i], b = grid[i + 6], c_ = grid[i + 12]; if (getColor(a) === getColor(b) && getColor(b) === getColor(c_)) matches.push([i, i + 6, i + 12]); } } return matches; } function monsterTurn() { isPlayerTurn = false; document.body.classList.add('enemy-turn'); setTimeout(() => { let moved = false; for (let i = 0; i < 36 && !moved; i++) { for (let j = i + 1; j < 36; j++) { let tmp = grid[i].className; grid[i].className = grid[j].className; grid[j].className = tmp; if (checkMatches().length > 0) { moved = true; setTimeout(() => cascadeAndApply(), 300); break; } else { grid[j].className = grid[i].className; grid[i].className = tmp; } } } if (!moved) startTurn(); }, 800 + Math.random() * 1000); } function showLog(name, combo, dmg, crit, chance, def, heal, isPlayer) { logBox.innerHTML = `
${name} собрал ${combo} комбо
Урон: ${dmg}${crit ? ' (КРИТ!)' : ''}
Щит: +${def}, Лечение: +${heal}
Крит. шанс: ${chance}%
`; } // Drag logic (touch) board.addEventListener("touchstart", e => { const t = e.touches[0]; const el = document.elementFromPoint(t.clientX, t.clientY); if (el && el.classList.contains('cell')) dragging = el; }); board.addEventListener("touchmove", e => { if (!dragging) return; const t = e.touches[0]; const el = document.elementFromPoint(t.clientX, t.clientY); if (el && el.classList.contains('cell') && el !== dragging) { let tmp = dragging.className; dragging.className = el.className; el.className = tmp; dragging = el; } }); board.addEventListener("touchend", () => { dragging = null; if (isPlayerTurn) { clearInterval(turnTimer); board.style.pointerEvents = 'none'; cascadeAndApply(); } });
